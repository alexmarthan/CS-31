#include <iostream>
using namespace std;

int main() {
    int len;
    cout << "Enter a number: ";
    cin >> len;
    
    for (int i = 0; i < len; i++){
        int j = i+1 ;
        while(j < len){
            j++;
            cout << " ";
        }
        cout << "#" << endl;
    }
    return(0);
}

#include <iostream>
using namespace std;

int main() {
    int len;
    cout << "Enter a number: ";
    cin >> len;
    
    if(len < 0){
        return(1);
    }
    
    int i = 0;
    do {
        
        int j = i+1 ;
        while(j < len){
            j++;
            cout << " ";
        }
        cout << "#" << endl;
        i++;
    } while(i < len);
    
    return(0);
}
